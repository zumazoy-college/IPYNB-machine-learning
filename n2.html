numeric_columns = ['view_count', 'duration', 'channel_follower_count'] #     

#   
stats_data = []
for col in numeric_columns:
    mean_val = df[col].mean() #  
    ci_val = 1.96 * df[col].std() / np.sqrt(len(df)) # 95%    : 1.96    / n
    stats_data.append({
        'attribute': col,
        'mean': mean_val,
        'ci': ci_val
    }) # c    

stats_df = pd.DataFrame(stats_data)

#  barplot
plt.figure(figsize=(12, 8)) #     128 
ax = sns.barplot(data=stats_df, # barplot      
                 x='attribute', #     X (        'attribute')
                 y='mean', #     Y (       'mean')
                 hue='attribute', #      (    'attribute'     )
                 palette=['skyblue', 'lightcoral', 'lightgreen'], # palette     
                 legend=False) # legend=False  ,         X

#    
for i, (_, row) in enumerate(stats_df.iterrows()): #     
    x_pos = i
    y_val = row['mean']
    ci_val = row['ci']
    
    #   
    ax.errorbar(x=x_pos, y=y_val, yerr=ci_val, # errorbar()    
                color='red', linewidth=2, capsize=5, capthick=2)
    
    #  
    ax.text(x_pos + 0.25, y_val + y_val*0.05,
            f'Mean: {y_val:.2e}\nCI: {ci_val:.2e}', 
            ha='center', va='bottom', fontsize=10, fontweight='bold',
            bbox=dict(boxstyle="round,pad=0.3", facecolor="white", alpha=0.8)) #

boxplot_stats = {}

for col in numeric_columns:
    data = df[col].dropna()
    Q1 = data.quantile(0.25)
    Q3 = data.quantile(0.75)
    IQR = Q3 - Q1
    
    #  
    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR
    outliers = data[(data < lower_bound) | (data > upper_bound)]
    
    boxplot_stats[col] = {
        'median': data.median(),
        'outliers_count': len(outliers)
    }

#  
max_median_attr = max(boxplot_stats.items(), key=lambda x: x[1]['median'])[0]
max_median_value = boxplot_stats[max_median_attr]['median']

#   
max_outliers_attr = max(boxplot_stats.items(), key=lambda x: x[1]['outliers_count'])[0]
max_outliers_count = boxplot_stats[max_outliers_attr]['outliers_count']

#   
min_outliers_attr = min(boxplot_stats.items(), key=lambda x: x[1]['outliers_count'])[0]
min_outliers_count = boxplot_stats[min_outliers_attr]['outliers_count']

print(f"    : {max_outliers_count}")
